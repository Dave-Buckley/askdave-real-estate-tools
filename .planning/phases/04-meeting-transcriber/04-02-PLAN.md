1. **Extend `src/main/ipc.ts`** -- Add transcriber IPC handlers inside `registerIPCHandlers()`. Add import at top: `import { createRecorderWindow, closeRecorderWindow, getRecorderWindow } from './transcriber'` and `import { getPanelWindow } from './tray'` (getPanelWindow may already be imported -- check).

Add these handlers at the end of registerIPCHandlers, before the closing `}`:

```typescript
  // --- Transcriber ---

  // Open the pop-out recorder window
  ipcMain.on('transcriber:open-recorder', () => {
    if (getRecorderWindow() && !getRecorderWindow()!.isDestroyed()) {
      getRecorderWindow()!.focus()
      return
    }
    createRecorderWindow()
  })

  // Close the pop-out recorder window
  ipcMain.on('transcriber:close-recorder', () => {
    closeRecorderWindow()
  })

  // Relay recording state from pop-out to panel
  ipcMain.on('transcriber:state-update', (_event, state: { state: string; elapsed?: number }) => {
    const panel = getPanelWindow()
    if (panel && !panel.isDestroyed()) {
      panel.webContents.send('transcriber:state', state)
    }
  })

  // Relay transcript result from pop-out to panel, then close pop-out
  ipcMain.on('transcriber:complete', (_event, transcript: string) => {
    const panel = getPanelWindow()
    if (panel && !panel.isDestroyed()) {
      panel.webContents.send('transcriber:state', { state: 'done', transcript })
    }
    // Close recorder window after relay
    setTimeout(() => closeRecorderWindow(), 100)
  })

  // Get whisper model preference
  ipcMain.handle('transcriber:get-model', () => {
    return store.get('whisperModel')
  })

  // Save whisper model preference
  ipcMain.handle('transcriber:set-model', (_event, modelId: string) => {
    store.set('whisperModel', modelId)
    return modelId
  })
```
